REM === Flipper Zero BadUSB Script: Hardware Report for macOS ===
REM === Gathers detailed information on system hardware ===
REM === Includes: CPU, RAM, storage devices, battery health, network adapters, connected USB and Bluetooth devices, screen dimensions, speaker type, and input devices (keyboard, mouse) ===
REM === Saves the report as "Hardware Report.txt" on the desktop ===
REM === Author: GhosTech-IT ===

REM === Open Spotlight ===
HOLD GUI
HOLD SPACE
RELEASE SPACE
RELEASE GUI
WAIT_FOR_BUTTON_PRESS

REM === Launch Terminal ===
STRING terminal
WAIT_FOR_BUTTON_PRESS
ENTER
WAIT_FOR_BUTTON_PRESS

REM === Position Terminal Center Screen ===
STRING osascript -e 'tell application "Terminal" to set bounds of front window to {400, 200, 1200, 800}'
WAIT_FOR_BUTTON_PRESS
ENTER
WAIT_FOR_BUTTON_PRESS

REM === Create Report Header ===
STRING echo "=== Full Hardware Report - $(date) ===" > ~/Desktop/"Hardware Report.txt"
WAIT_FOR_BUTTON_PRESS
ENTER
WAIT_FOR_BUTTON_PRESS

REM === System Overview ===
STRING echo "=== System Overview ===" >> ~/Desktop/"Hardware Report.txt"
WAIT_FOR_BUTTON_PRESS
ENTER
STRING system_profiler SPHardwareDataType >> ~/Desktop/"Hardware Report.txt"
WAIT_FOR_BUTTON_PRESS
ENTER
WAIT_FOR_BUTTON_PRESS
